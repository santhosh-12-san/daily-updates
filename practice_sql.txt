1 Create and Select Database
SHOW DATABASES;
CREATE DATABASE mydb;
USE mydb;
DROP DATABASE mydb;

2️ Create , Alter or Drop Table
CREATE TABLE students(
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(50),
    age INT,
    marks INT
);

ALTER TABLE students ADD address VARCHAR(100);
ALTER TABLE students MODIFY marks FLOAT;
ALTER TABLE students DROP address;

DROP TABLE students;

3️ Insert , Update and Delete
INSERT INTO students(name, age, marks)
VALUES ('Rahul', 20, 85);

UPDATE students SET marks = 90 WHERE id = 1;

DELETE FROM students WHERE id = 1;

4️ Select
SELECT * FROM students;
SELECT name, marks FROM students;
SELECT DISTINCT name FROM students;

5️ WHERE Conditions
SELECT * FROM students WHERE age = 20;
SELECT * FROM students WHERE marks > 80;
SELECT * FROM students WHERE age BETWEEN 18 AND 25;
SELECT * FROM students WHERE name LIKE 'A%';
SELECT * FROM students WHERE name IN ('Rahul', 'Ramesh');

6️ AND , OR and NOT
SELECT * FROM students WHERE age > 18 AND marks > 80;
SELECT * FROM students WHERE age < 18 OR marks > 90;
SELECT * FROM students WHERE NOT age = 20;

7️ CASE Conditional Output
SELECT name,
       marks,
       CASE
           WHEN marks >= 90 THEN 'A'
           WHEN marks >= 75 THEN 'B'
           ELSE 'C'
       END AS grade
FROM students;

8️ Aggregate Functions
SELECT COUNT(*), AVG(marks), MIN(marks), MAX(marks), SUM(marks)
FROM students;

9️ GROUP BY
SELECT age, COUNT(*) FROM students
GROUP BY age;

10  HAVING with GROUP BY
SELECT age, COUNT(*) 
FROM students 
GROUP BY age
HAVING COUNT(*) > 2;

1️1️ ORDER BY
SELECT * FROM students ORDER BY marks DESC;
SELECT * FROM students ORDER BY name ASC;

1️2️ LIMIT
SELECT * FROM students LIMIT 5;
SELECT * FROM students ORDER BY marks DESC LIMIT 3;

1️3️ INNER JOIN
SELECT s.name, c.course_name
FROM students s
INNER JOIN courses c ON s.id = c.student_id;

1️4️ LEFT JOIN
SELECT s.name, c.course_name
FROM students s
LEFT JOIN courses c ON s.id = c.student_id;

1️5️ RIGHT JOIN
SELECT s.name, c.course_name
FROM students s
RIGHT JOIN courses c ON s.id = c.student_id;

16 FULL OUTER JOIN MySQL uses UNION
SELECT * FROM students
LEFT JOIN courses ON students.id = courses.student_id
UNION
SELECT * FROM students
RIGHT JOIN courses ON students.id = courses.student_id;

1️7️ Simple Subquery
SELECT * FROM students
WHERE marks > (SELECT AVG(marks) FROM students);


1️8️ Primary Key and Foreign Key
CREATE TABLE departments(
    dept_id INT PRIMARY KEY,
    dept_name VARCHAR(50)
);

CREATE TABLE employees(
    emp_id INT PRIMARY KEY,
    name VARCHAR(50),
    dept_id INT,
    FOREIGN KEY (dept_id) REFERENCES departments(dept_id)
);

ALTER TABLE employees ADD FOREIGN KEY (dept_id) REFERENCES departments(dept_id);

Deleting With Condition
DELETE FROM students WHERE marks < 40;

Find Duplicates
SELECT name, COUNT(name)
FROM students
GROUP BY name
HAVING COUNT(name) > 1;

Even and Odd Rows
SELECT * FROM students WHERE id % 2 = 0;   
SELECT * FROM students WHERE id % 2 = 1; 

LIKE Patterns
SELECT * FROM students WHERE name LIKE 'A%';   
SELECT * FROM students WHERE name LIKE '%h';  
SELECT * FROM students WHERE name LIKE '%an%';
